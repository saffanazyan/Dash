function displayResourceCharts(dataArray) { const chartsContainer = document.getElementById('chartsContainer'); // Extract dt_infotime and s_attemp values from each object const dtInfotimeValues = dataArray.map(obj => obj.dt_infotime); const sAttempValues = dataArray.map(obj => obj.s_attemp); // Create a line chart for dt_infotime const dtInfotimeChartContainer = document.createElement('div'); dtInfotimeChartContainer.style.marginBottom = '20px'; dtInfotimeChartContainer.innerHTML = ``; chartsContainer.appendChild(dtInfotimeChartContainer); // Get the canvas element for dt_infotime const dtInfotimeCtx = document.getElementById('dtInfotimeChart').getContext('2d'); // Create a line chart for dt_infotime new Chart(dtInfotimeCtx, { type: 'line', data: { labels: Array.from({ length: dtInfotimeValues[0].length }, (_, i) => i + 1), datasets: dtInfotimeValues.map((values, index) => ({ label: `dt_infotime (Object ${index + 1})`, data: values, borderColor: `rgba(${Math.floor(Math.random() * 256)}, ${Math.floor(Math.random() * 256)}, ${Math.floor(Math.random() * 256)}, 1)`, borderWidth: 2, fill: false })) }, options: { responsive: true, maintainAspectRatio: false, scales: { x: { type: 'category', title: { display: true, text: 'Data Points' } }, y: { title: { display: true, text: 'Values' } } } } }); // Create a line chart for s_attemp const sAttempChartContainer = document.createElement('div'); sAttempChartContainer.style.marginBottom = '20px'; sAttempChartContainer.innerHTML = ``; chartsContainer.appendChild(sAttempChartContainer); // Get the canvas element for s_attemp const sAttempCtx = document.getElementById('sAttempChart').getContext('2d'); // Create a line chart for s_attemp new Chart(sAttempCtx, { type: 'line', data: { labels: Array.from({ length: sAttempValues[0].length }, (_, i) => i + 1), datasets: sAttempValues.map((values, index) => ({ label: `s_attemp (Object ${index + 1})`, data: values, borderColor: `rgba(${Math.floor(Math.random() * 256)}, ${Math.floor(Math.random() * 256)}, ${Math.floor(Math.random() * 256)}, 1)`, borderWidth: 2, fill: false })) }, options: { responsive: true, maintainAspectRatio: false, scales: { x: { type: 'category', title: { display: true, text: 'Data Points' } }, y: { title: { display: true, text: 'Values' } } } } }); }
